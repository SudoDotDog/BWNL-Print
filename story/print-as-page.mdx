import { printReactElementAsPage } from "../src/index";
import { Card } from "antd";
import { Playground } from "docz";

# Print As Page

## Simple

<Playground>
    <button onClick={() => {
        printReactElementAsPage(<html>
            <head></head>
            <body>
                <div>Hello There</div>
            </body>
        </html>);
    }}>Print</button>
</Playground>

## Inject CSS

<Playground>
    <button onClick={() => {
        printReactElementAsPage((<html>
            <head>
                <link 
                    href="https://cdn.jsdelivr.net/npm/antd@4.0.1/dist/antd.min.css"
                    rel="stylesheet"
                    type="text/css"
                />
            </head>
            <body>
                <Card
                title="Hello"
                >
                    Hello There
                </Card>
            </body>
        </html>), {
            needLoads: true,
        });
    }}>Print</button>
</Playground>
