import { printReactElementAsPage } from "../src/index";
import { Card } from "antd";
import { Playground } from "docz";

# Print As Page

## Simple

<Playground>
    <button onClick={() => {
        printReactElementAsPage(<div>Hello There</div>);
    }}>Print</button>
</Playground>

## Inject CSS

<Playground>
    <button onClick={() => {
        printReactElementAsPage((<Card
            title="Hello"
        >
            Hello There
        </Card>), {
            injectCSSFiles: [
                'https://cdn.jsdelivr.net/npm/antd@4.0.1/dist/antd.min.css',
            ],
        });
    }}>Print</button>
</Playground>

## Inject CSS Working on Safari

<Playground>
    <button onClick={() => {
        printReactElementAsPage((<Card
            title="Hello"
        >
            Hello There
        </Card>), {
            injectCSSFiles: [
                'https://cdn.jsdelivr.net/npm/antd@4.0.1/dist/antd.min.css',
            ],

        });
    }}>Print</button>
</Playground>
